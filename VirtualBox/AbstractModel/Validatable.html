def validates_presence_of(field)
  if field.is_a?(Array)
    field.map { |v| validates_presence_of(v) }.all? { |v| v == true }
  else
    value = send(field)
    if value.nil? || value == ""
      add_error(field, "must not be blank.")
      return false
    else
      return true
    end
  end
end
